runbash

github.com/cadorn/runbash cadorn.github.io/runbash

Run bash and bash.origin from NodeJS. Easily.

Install

npm install runbash

Usage

const RUNBASH = require("runbash");

RUNBASH([
    "echo 'Hello World'",
    "echo 'FOO: BAR'"
], {
    verbose: false,
    progress: false,
    wrappers: {
        "bash.origin": true
    },
    wait: true,
    exports: true,    
}).then(function (result) {

    // result.code
    // result.stdout
    // result.status
    // result.exports.FOO

    return null;
});
